Golang RPC

Before we can talk about Remote Procedure Calls, we first need to define how a process can be bound to a particular TCP (or UDP) port. Go provides very simple interfaces to make this task trivial: 

import(
    "log"
    "net"
    "net/rpc"
)

func Run(s *Server) {
    address, err := net.ResolveTCPAddr("tcp", s.Port)
    if err != nil {
        log.Fatal(err)
    }

    inbound, err := net.ListenTCP("tcp", address)
    if err != nil {
        log.Fatal(err)
    }

    rpc.Register(s)
    rpc.Accept(inbound)
}
